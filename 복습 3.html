< Object에 함수 넣기 >
var SetC = { //SetcC라는 변수에 객체를 담고, 해당 객체에 LinkSetC 등의 Property(Method) 입력
Link : function (color){
var i = 0;
var a = document.querySelectorAll('a');
while(i< a.length){
a[i].style.color = color;
i = i + 1;}
},
Body : function (color){
document.querySelector('body').style.color = color
},

Bodybg : function (color){
document.querySelector('body').style.backgroundColor = color
}
}

function nightNdayHandler(self){
if(self.value === 'night'){
SetC.Body('navy');
SetC.Bodybg('pink');
SetC.Link('yellow');
self.value = 'day'
}
else{
SetC.Body('darkgreen');
SetC.Bodybg('lightgrey');
SetC.Link('orange');
self.value = 'night'
}
}
* ctrl-alt-방향키 : 처음 커서 위치한 자리에 다중입력 커서 생성
Handler 함수도 객체에 넣음.
